CREATE TABLE Department
(
  Deptno NUMERIC(2) PRIMARY KEY,
  Deptname VARCHAR(20) UNIQUE,
  Location VARCHAR(20)
);


-- Question 3
CREATE TABLE Employee
(

    Empno NUMERIC(3) PRIMARY KEY,
    ManagerID NUMERIC(3),
    FirstName VARCHAR(20) NOT NULL,
    LastName VARCHAR(20),
    UserID VARCHAR(20),
    Deptno NUMERIC(2)
    CONSTRAINT fk_deptno 
    FOREIGN KEY(Deptno)
    REFERENCES Department(Deptno)
    ON DELETE CASCADE,
    Salary NUMERIC(7,2),
    Commission NUMERIC(3),
    JoiningDate DATE,
    Designation VARCHAR(25)

);

CREATE TABLE [Order]
(

    OrderNo NUMERIC(2) PRIMARY KEY,

        FOREIGN KEY (CustNo)

CREATE TABLE Course

    Fees DECIMAL,

-- Question 14

CREATE TABLE Batch
(

    BatchId VARCHAR(30) PRIMARY KEY,
    CourseId VARCHAR(5)
        CONSTRAINT fk_courseid
        FOREIGN KEY (CourseId)
        REFERENCES Course(CourseId),
    BatchName VARCHAR(1),
);

-- Question 15

CREATE TABLE Student
(

    StudId VARCHAR(5) PRIMARY KEY,
    LastName VARCHAR(25),
    MiddleName VARCHAR(30),
    FirstName VARCHAR(20),
    Dob DATE DEFAULT GETDATE(),
    BatchId VARCHAR(5)
        CONSTRAINT fk_batchid
        FOREIGN KEY (BatchId)
        REFERENCES Batch(BatchId),

    StudentAddress VARCHAR(50),
    City VARCHAR(20),
    StudentState VARCHAR(2),
    Zipcode VARCHAR(9),
    Telephone VARCHAR(10),
    Fax VARCHAR(10),
    Email VARCHAR(30),
    Grade VARCHAR(1)
        CONSTRAINT chk_grade
        CHECK(Grade IN('A','A+','A-',
                       'B','B+','B-',
                       'C','C+','C-',
                       'D','D+','D-',
                       'F','F+','F-')),
);
